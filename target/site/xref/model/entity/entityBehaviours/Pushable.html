1   package model.entity.entityBehaviours;

3   import model.entity.Empty;
4   import model.entity.Entity;
5   
6   /**
7    * The pushable class
8    *
9    * @author Luc anthoni
10   * @version 1.0
11   */
12  public class Pushable
13  {
14      /**
15       * The entity variable is used to allow the fallable class access to the variables of entity such as x, y, or map
16       */
17      private Entity entity;
18  
19      /**
20       *
21       * @param entity The entity that you want to make pushable<
22       */
23      public Pushable(Entity entity) {
24          this.entity = entity;
25      }
26  
27      /**
28       * The method only checks if the pushable object can move right and moves it if it can
29       */
30      public void pushRight() {
31          entity.map[entity.x][entity.y] = new Empty(entity.x, entity.y, entity.map);
32          entity.x++;
33          entity.map[entity.x][entity.y] = entity;
34      }
35  
36      /**
37       * The method only checks if the pushable object can move left and moves it if it can
38       */
39      public void pushLeft() {
40          entity.map[entity.x][entity.y] = new Empty(entity.x, entity.y, entity.map);
41          entity.x--;
42          entity.map[entity.x][entity.y] = entity;
43      }
44  
45  }
