20              entity.map[entity.x][entity.y] = new Empty(entity.x, entity.y, entity.map);

22              setFalling(true);
23              entity.map[entity.x][entity.y] = entity;
24          }
25          else {entity.map[entity.x][entity.y].fallable.setFalling(false);}
26          if (entity.map[entity.x][entity.y + 1].getSprite() == 'p')
27          {
28              if (entity.map[entity.x][entity.y].fallable.falling == true)
29              {
30                  BoardFrame.kill(); //kill from BoardFrame

32          }
